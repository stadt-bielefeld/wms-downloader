        start(this, options, callback);
    }

    /**
   * Cancels a download.
   * @param {string} id ID of the task
   * @param {function} callback Callback function like `function(err){}`
   */
    cancel(id, callback) {
        cancel(this, id, callback);
    }

    /**
   * Returns the progress of a download.
   * @param {string} id 
   * @returns {object} aaa
   */

        return getProgress(this, id);
